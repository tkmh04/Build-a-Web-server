__________________
server {
    listen 80;
    server_name 192.168.168.20;
    root /var/www/html;
    index index.php index.html index.htm;

location / {
    # Dòng này tương ứng với RewriteRule trong .htaccess của bạn
    # Nó sẽ lấy phần sau dấu / và gán vào biến url
    try_files $uri $uri/ /index.php?url=$uri&$args;
}
    # Đoạn này để Nginx biết cách chạy file PHP
    location ~ \.php$ {
        include snippets/fastcgi-php.conf;
        # Kiểm tra kỹ số 8.2 (hoặc 7.4) cho đúng bản PHP của bạn
        fastcgi_pass unix:/run/php/php8.2-fpm.sock;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_nam>
        include fastcgi_params;
    }

    # Chặn truy cập vào các file cấu hình nhạy cảm
    location ~ /\.ht {
        deny all;
    }
}
