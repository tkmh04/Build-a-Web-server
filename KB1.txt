#!/usr/sbin/nft -f

# 1. Xóa toàn bộ quy tắc cũ để tránh xung đột
flush ruleset

table inet firewall {
    # Định nghĩa danh sách trắng (1 giờ)
    set whitelist {
        type ipv4_addr
        flags dynamic
        timeout 1h
    }

    # Định nghĩa danh sách đen (10 phút)
    set blacklist {
        type ipv4_addr
        flags dynamic
        timeout 10m
    }

    chain input {
        type filter hook input priority 0; policy drop;

        # Chấp nhận các kết nối đã thiết lập và localhost
        ct state established,related accept
        iif "lo" accept

        # Chặn ngay lập tức nếu IP nằm trong blacklist
        ip saddr @blacklist drop

        # Cho phép IP trong whitelist
        ip saddr @whitelist accept

        # Cho phép các cổng dịch vụ cơ bản
        tcp dport { 22, 80, 443 } accept

        # Giới hạn ICMP (Ping) để tránh flood
        ip protocol icmp limit rate 5/second accept

        # XỬ LÝ LỖI DÒNG SYN & UDP:
        # Nếu vượt quá 20 gói SYN/giây tới port 80 -> Cho vào blacklist và drop
        tcp flags syn tcp dport 80 limit rate over 20/second update @blacklist { ip saddr } drop
        tcp flags syn tcp dport 80 accept

        # Nếu vượt quá 10 gói UDP/giây tới port 80 -> Cho vào blacklist và drop
        udp dport 80 limit rate over 10/second update @blacklist { ip saddr } drop
        udp dport 80 accept
    }

    chain forward {
        type filter hook forward priority 0; policy drop;
        ct state established,related accept
        
        # Cho phép dải mạng nội bộ (nếu có)
        ip saddr 192.168.168.0/24 accept
    }

    chain output {
        # Bổ sung đầy đủ thông số cho chain output
        type filter hook output priority 0; policy accept;
    }
}
